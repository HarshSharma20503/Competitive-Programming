template <typename T>
T Power_MOD(T a,T b)
{
	T res=1;
	while(b>0)
	{
		if(b&1)res=(res*a)%MOD;
		a=(a*a)%MOD;
		b>>=1;
	}

}
template <typename T>
T Power(T a,T b)
{
	long long res=1;
	while(b>0)
	{
		if(b&1)
		{
			res=res*a;
		}
		a*=a;
		b>>=1;
	}
	return res;
}
